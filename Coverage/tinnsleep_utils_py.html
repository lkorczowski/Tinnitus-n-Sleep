5def epoch(a, size, interval, axis=-1): 

6 """ Small proof of concept of an epoching function using NumPy strides 

7 License: BSD-3-Clause 

8 Copyright: David Ojeda <david.ojeda@gmail.com>, 2018. 

9 Modified by Louis Korczowski <louis.korczowski@gmail.com>, 2020. 

10 Create a view of `a` as (possibly overlapping) epochs. 

11 The intended use-case for this function is to epoch an array representing 

12 a multi-channels signal with shape `(n_samples, n_channels)` in order 

13 to create several smaller views as arrays of size `(size, n_channels)`, 

14 without copying the input array. 

15 This function uses a new stride definition in order to produce a view of 

16 `a` that has shape `(num_epochs, ..., size, ...)`. Dimensions other than 

17 the one represented by `axis` do not change. 

18 

19 Parameters 

20 ---------- 

21 a: array_like 

22 Input array 

23 size: int 

24 Number of elements (i.e. samples) on the epoch. 

25 interval: int 

26 Number of elements (i.e. samples) to move for the next epoch. 

27 axis: int 

28 Axis of the samples on `a`. For example, if `a` has a shape of 

29 `(num_observation, num_samples, num_channels)`, then use `axis=1`. 

30 Returns 

31 ------- 

32 ndarray 

33 Epoched view of `a`. Epochs are in the first dimension. 

34 """ 

35 a = np.asarray(a) 

36 a = check_array(a) 

37 if (size < 1) | (interval < 1): 

38 raise ValueError("Invalid range for parameters") 

39 

40 n_samples = a.shape[axis] 

41 n_epochs = (n_samples - size) // interval + 1 

42 

43 new_shape = list(a.shape) 

44 new_shape[axis] = size 

45 new_shape = (n_epochs,) + tuple(new_shape) 

46 

47 new_strides = (a.strides[axis] * interval,) + a.strides 

48 

49 return stride_tricks.as_strided(a, new_shape, new_strides)

53 return int(np.ceil((N-T+1) / I))
